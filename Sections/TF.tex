\section{Common True/False Questions}

\begin{prob}{}{}
Prove or disprove: Suppose that $X = U \cup V$ where $U$ and $V$ are both open and simply connected. Then, $X$ is simply connected.
\end{prob}

\begin{prob}{}{}
    Prove or disprove: If $f: X \to Y$ is continuous and surjective, then the induced homeomorphism $f_*: \pi_1(X) \to \pi_1(Y)$ is surjective.
\end{prob}

\begin{prob}{}{}
    Prove or disprove: If $f: X \to Y$ is continuous and injective, then the induced homeomorphism $f_*: \pi_1(X) \to \pi_1(Y)$ is injective.
\end{prob}

\begin{prob}{}{}
    Prove or disprove: Let $X$ be a compact topological space and $\{F_n\}$ a nested sequence of closed sets $F_1 \supseteq F_2 \supseteq \cdots$. Then $\cap F_n \neq \varnothing$.
\end{prob}

\begin{prob}{}{}
    Prove or disprove: Let $X$ be a compact topological space and $\{U_n\}$ a nested sequence of open sets $U_1 \sq U_2 \sq \cdots$. Then $\cap U_n \neq \varnothing$.
\end{prob}

\begin{prob}{}{}
    Prove or disprove: A closed and bounded subset of a topological space is compact.
\end{prob}

\begin{prob}{}{}
    Prove or disprove: The continuous image of a closed set is closed.
\end{prob}

\begin{prob}{}{}
    Prove or disprove: If $f: X \to Y$ is a continuous surjective and $Y$ is Hausdorff, then $X$ is Hausdorff.
\end{prob}
